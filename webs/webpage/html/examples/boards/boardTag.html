html 태그로 만드는 게시판입니다

<da include="boardlist.board" type="boardlist" dbpath="example.dadb" code="da_board" brdid="1" renderview="BoardView" rendertype="table" title="table type"> </da>

태그속성
include="삽입하려는 파일이름"
type="게시판의경우 boardlist(일반게시판) 와 grouplist(그룹게시판) 가있음"
dbpath="sqlite 디비의 디비경로이름(/html/admin/config/DbandFile.html 에서 디비작업)"
code="테이블이름"
brdid="게시판 아이디(정수형 아이디가 다르면 다른게시판)"
renderview="실제 게시판을 보여주는 클래스이름(/aMvc/base/viewLibBase.js 나 templateBase.js 에구현"
rendertype="게시판의 종류(table, media, thumbnail, thumbnailR, paintthumbnail, timeline, selectlist_h, selectlist_v)를 설정"
title="게시판 제목표시"
